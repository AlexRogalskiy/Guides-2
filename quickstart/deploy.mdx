## Bringing it all together

Here's what your code should look like so far.

```jsx
import { FlatfileClient } from '@flatfile/api';
import { Client, FlatfileVirtualMachine FlatfileEvent} from '@flatfile/listener';
import { FlatfileRecord } from '@flatfile/hooks';
import { recordHook } from '@flatfile/plugin-record-hook';
import * as EmailValidator from 'email-validator'

void main();

async function main() {
  const flatfile = new FlatfileClient({
    publishableKey: '1234', //find this at dashboard.flatfile.com
  });

  const blueprint = {
      sheets: [
        {
          name: 'My Sheet 1',
          description: 'hello',
          slug: 'mysheet1',
          fields: [
            {
              key: 'first_name',
              type: 'string',
              label: 'First Name',
              description: 'The first name of the customer',
              constraints: [
                { type: 'required' },
              ],
            },
            {
              key: 'last_name',
              type: 'string',
              label: 'Last Name',
              description: 'The last name of the customer',
              constraints: [
                { type: 'required' },
              ],
            },
            {
              key: 'full_name',
              type: 'string',
              label: 'Full Name',
              description: 'The full name of the customer',
            },
          ],
        },
      ],
    }

  // Create a Workbook
  const createWorkbook = await flatfile.workbooks.create({
    name: 'My Workbook 1',
    blueprint: blueprint
  });

  flatfile.use(recordHook('mysheet1', (record: FlatfileRecord) => {
    record.transformIfPresent(['first_name', 'last_name'], 'full_name', () => `${first_name} ${last_name}`)
	record.validate('email', 'Email is invalid',  (email) => EmailValidator.validate(email))
	return record;
  }));

}
```
