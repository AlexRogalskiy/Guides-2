## Adding your first action 

In this example, we're going to add an action to the primary action `submitAction` you see in the workbook you have already deployed. 
This action will take data from the sheet and submit it to webhook.site.

```jsx
import { FlatfileEvent } from '@flatfile/listener'

const { URL } = require('url')
const https = require('https')
const http = require('http')
function makeHttpRequest(params) {
  const { pathname, body, hostname, protocol, port } = params
  const options = {
    protocol: protocol || 'https:',
    hostname: hostname || 'platform.flatfile.com/api',
    method: 'POST',
    ...(port ? { port } : {}),
    path: pathname,
  }
  const request = protocol === 'http:' ? http : https
  return new Promise((resolve, reject) => {
    const req = request.request(options, (res) => {
      let body = ''
      res.on('data', (chunk) => {
        body += chunk
      })
      res.on('end', () => {
        if (res.statusCode === 200) {
          resolve(body)
        } else {
          resolve('Failure')
        }
      })
      res.on('error', () => {
        console.log('error')
        reject(Error('HTTP call failed'))
      })
    })
    // The below 2 lines are most important part of the whole snippet.
    req.write(JSON.stringify(body))
    req.end()
  })
}

export const webhookAction = async (
  event,
  webhookUrl
) => {

  const fullUrl = new URL(webhookUrl)
  try {
    const body = {
      event,
    }

    const { protocol, hostname, port, pathname } = fullUrl
    await makeHttpRequest({
      pathname,
      body,
      hostname,
      protocol,
      port,
    })
  } catch (error) {
    console.log(`NodeHttpsAction[error]: ${JSON.stringify(error, null, 2)}`)
  }
}

export default function(listener) {

  listener.on('action:triggered', async (event) => {
    await webhookAction(event, 'https://webhook.site/3f74fe0a-0899-48ad-bbde-4571aaaf7ec6')
  })

}
```
